Salt (ntp.sls):
ntp_pkg: pkg.installed: [ntp]
ntp_config: file.managed: /etc/ntp.conf
ntp_service: service.running: ntp
Ansible (ntp/tasks/main.yml):
- name: Install ntp
  apt: name=ntp state=present
- name: Deploy ntp.conf
  template: src=ntp.conf.j2 dest=/etc/ntp.conf
- name: Start ntp service
  systemd: name=ntp state=started enabled=yes
